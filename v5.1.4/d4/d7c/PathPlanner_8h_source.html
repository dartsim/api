90 bool PathPlanner<R>::planPath(dynamics::Skeleton* robot, const std::vector<size_t> &dofs,
91  const std::vector<Eigen::VectorXd> &start, const std::vector<Eigen::VectorXd> &goal,
92  std::list<Eigen::VectorXd> &path) {
93 
94  Eigen::VectorXd savedConfiguration = robot->getPositions(dofs);
95 
96  // ====================================================================
97  // Check for collisions in the start and goal configurations
98 
99  // Sift through the possible start configurations and eliminate those that are in collision
100  std::vector<Eigen::VectorXd> feasibleStart;
101  for(unsigned int i = 0; i < start.size(); i++) {
102  robot->setPositions(dofs, start[i]);
103  if(!world->checkCollision()) feasibleStart.push_back(start[i]);
104  }
105 
106  // Return false if there are no feasible start configurations
107  if(feasibleStart.empty()) {
108  printf("WARNING: PathPlanner: Feasible start points are empty!\n");
109  return false;
110  }
111 
112  // Sift through the possible goal configurations and eliminate those that are in collision
113  std::vector<Eigen::VectorXd> feasibleGoal;
114  for(unsigned int i = 0; i < goal.size(); i++) {
115  robot->setPositions(dofs, goal[i]);
116  if(!world->checkCollision()) feasibleGoal.push_back(goal[i]);
117  }
118 
119  // Return false if there are no feasible goal configurations
120  if(feasibleGoal.empty()) {
121  printf("WARNING: PathPlanner: Feasible goal points are empty!\n");
122  return false;
123  }
124 
125  // ====================================================================
126  // Make the correct RRT algorithm for the given method
127 
128  // Direct the search towards single or bidirectional
129  bool result = false;
130  if(bidirectional)
131  result = planBidirectionalRrt(robot, dofs, feasibleStart, feasibleGoal, path);
132  else {
133  if(feasibleGoal.size() > 1) fprintf(stderr, "WARNING: planPath is using ONLY the first goal!\n");
134  result = planSingleTreeRrt(robot, dofs, feasibleStart, feasibleGoal.front(), path);
135  }
136 
137  // Restore previous robot configuration
138  robot->setPositions(dofs, savedConfiguration);
139 
140  return result;
141 }

190 bool PathPlanner<R>::planBidirectionalRrt(dynamics::Skeleton* robot, const std::vector<int> &dofs,
191  const std::vector<Eigen::VectorXd> &start, const std::vector<Eigen::VectorXd> &goal,
192  std::list<Eigen::VectorXd> &path) {
193 
194  const bool debug = false;
195 
196  // Initialize both the start and goal RRTs.
197  // NOTE: We use the pointers for the RRTs to swap their roles in extending towards a target
198  // (random or goal) node.
199  start_rrt = new R(world, robot, dofs, start, stepSize);
200  goal_rrt = new R(world, robot, dofs, goal, stepSize);
201  R* rrt1 = start_rrt;
202  R* rrt2 = goal_rrt;
203 
204  // Expand the tree until the trees meet or the max # nodes is passed
205  double smallestGap = std::numeric_limits<double>::infinity();
206  size_t numNodes = rrt1->getSize() + rrt2->getSize();
207  while(numNodes < maxNodes) {
208 
209  // Swap the roles of the two RRTs. Remember, the first rrt reaches out to a target node and
210  // creates a new node and the second rrt reaches to _the new node_.
211  R* temp = rrt1;
212  rrt1 = rrt2;
213  rrt2 = temp;
214 
215  // Get the target node based on the bias
216  Eigen::VectorXd target;
217  double randomValue = ((double) rand()) / RAND_MAX;
218  if(randomValue < goalBias) target = goal[0];
219  else target = rrt1->getRandomConfig();
220 
221  // Based on the method, rrt1 either attempt to connect to the target directly or takes a step
222  if(connect) rrt1->connect(target);
223  else rrt1->tryStep(target);
224 
225  // rrt2 uses the last added node of rrt1 as a target and reaches out to it (connect or step)
226  // NOTE: If a node was not added, the nearest neighbor to the random node in rrt1 is used.
227  // NOTE: connect(x) and tryStep(x) functions return true if rrt2 can add the given node
228  // in the tree. In this case, this would imply that the two trees meet.
229  bool treesMet = false;
230  const Eigen::VectorXd& rrt2target = *(rrt1->configVector[rrt1->activeNode]);
231  if(connect) treesMet = rrt2->connect(rrt2target);
232  else treesMet = (rrt2->tryStep(rrt2target) == R::STEP_REACHED);
233 
234  // Check if the trees have met and create the path, if so.
235  if(treesMet) {
236  start_rrt->tracePath(start_rrt->activeNode, path);
237  goal_rrt->tracePath(goal_rrt->activeNode, path, true);
238  return true;
239  }
240 
241  // Update the number of nodes in the two trees
242  numNodes = rrt1->getSize() + rrt2->getSize();
243 
244  // Print the gap between the trees in debug mode
245  if(debug) {
246  double gap = rrt2->getGap(*(rrt1->configVector[rrt1->activeNode]));
247  if(gap < smallestGap) {
248  smallestGap = gap;
249  std::cout << "Gap: " << smallestGap << " Sizes: " << start_rrt->configVector.size()
250  << "/" << goal_rrt->configVector.size() << std::endl;
251  }
252  }
253  }
254 
255  // Maximum # of iterations are reached and path is not found - failed.
256  return false;
257 }
