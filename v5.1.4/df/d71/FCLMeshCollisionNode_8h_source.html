130 inline bool FCLMeshCollisionNode::EFtest(const fcl::Vec3f& p0,
131  const fcl::Vec3f& p1,
132  const fcl::Vec3f& r1,
133  const fcl::Vec3f& r2,
134  const fcl::Vec3f& r3,
135  fcl::Vec3f* p)
136 {
137  double ZERO1 = 0.00000001;
138  fcl::Vec3f n = (r3 - r1).cross(r2 - r1);
139  double s = (p1 - p0).dot(n);
140  if (std::abs(s) < ZERO1) return false;
141  double t = (r1 - p0).dot(n) / s;
142  fcl::Vec3f tmp = p0 + (p1 - p0) * t;
143  if (t >= 0
144  && t <= 1
145  && (((tmp - r1).cross(r2 - r1)).dot(n) >= 0)
146  && (((tmp - r2).cross(r3 - r2)).dot(n) >= 0)
147  && (((tmp - r3).cross(r1 - r3)).dot(n) >= 0)
148  )
149  {
150  *p = tmp;
151  return true;
152  }
153  return false;
154 }

229 inline double FCLMeshCollisionNode::triArea(
230  fcl::Vec3f p1, fcl::Vec3f p2, fcl::Vec3f p3)
231 {
232  fcl::Vec3f a = p2 - p1;
233  fcl::Vec3f b = p3 - p1;
234  double aMag = a[0] * a[0] + a[1] * a[1] + a[2] * a[2];
235  double bMag = b[0] * b[0] + b[1] * b[1] + b[2] * b[2];
236  double dp = a[0] * b[0] + a[1] * b[1] + a[2] * b[2];
237  double area = 0.5 * sqrt(aMag * bMag - dp * dp);
238  return area;
239 }
