349 template <size_t DOF>
350 void MultiDofJoint<DOF>::setCommand(size_t _index, double _command)
351 {
352  if (_index >= getNumDofs())
353  {
354  MULTIDOFJOINT_REPORT_OUT_OF_RANGE(setCommand, _index);
355  }
356 
357  switch (mJointP.mActuatorType)
358  {
359  case FORCE:
360  mCommands[_index] = math::clip(_command,
361  mMultiDofP.mForceLowerLimits[_index],
362  mMultiDofP.mForceUpperLimits[_index]);
363  break;
364  case PASSIVE:
365  if(0.0 != _command)
366  {
367  dtwarn << "[MultiDofJoint::setCommand] Attempting to set a non-zero ("
368  << _command << ") command for a PASSIVE joint [" << getName()
369  << "].\n";
370  }
371  mCommands[_index] = _command;
372  break;
373  case SERVO:
374  mCommands[_index] = math::clip(_command,
375  mMultiDofP.mVelocityLowerLimits[_index],
376  mMultiDofP.mVelocityUpperLimits[_index]);
377  break;
378  case ACCELERATION:
379  mCommands[_index] = math::clip(_command,
380  mMultiDofP.mAccelerationLowerLimits[_index],
381  mMultiDofP.mAccelerationUpperLimits[_index]);
382  break;
383  case VELOCITY:
384  mCommands[_index] = math::clip(_command,
385  mMultiDofP.mVelocityLowerLimits[_index],
386  mMultiDofP.mVelocityUpperLimits[_index]);
387  // TODO: This possibly makes the acceleration to exceed the limits.
388  break;
389  case LOCKED:
390  if(0.0 != _command)
391  {
392  dtwarn << "[MultiDofJoint::setCommand] Attempting to set a non-zero ("
393  << _command << ") command for a LOCKED joint [" << getName()
394  << "].\n";
395  }
396  mCommands[_index] = _command;
397  break;
398  default:
399  assert(false);
400  break;
401  }
402 }

1996 void MultiDofJoint<DOF>::updateTotalForceDynamic(
1997  const Eigen::Vector6d& _bodyForce,
1998  double _timeStep)
1999 {
2000  // Spring force
2001  const Eigen::Matrix<double, DOF, 1> springForce
2002  = (-mMultiDofP.mSpringStiffnesses).asDiagonal()
2003  *(getPositionsStatic() - mMultiDofP.mRestPositions
2004  + getVelocitiesStatic()*_timeStep);
2005 
2006  // Damping force
2007  const Eigen::Matrix<double, DOF, 1> dampingForce
2008  = (-mMultiDofP.mDampingCoefficients).asDiagonal()*getVelocitiesStatic();
2009
